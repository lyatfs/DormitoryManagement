cmake_minimum_required(VERSION 3.10)

project(DormitoryManagement LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include(FetchContent)

set(FETCHCONTENT_UPDATES_DISCONNECTED ON)
set(FETCHCONTENT_QUIET OFF)

FetchContent_Declare(ftxui
  GIT_REPOSITORY https://github.com/arthursonzogni/ftxui.git
  GIT_TAG        v5.0.0
  GIT_PROGRESS   TRUE
  GIT_SHALLOW    TRUE
  EXCLUDE_FROM_ALL
)
FetchContent_MakeAvailable(ftxui)

add_executable(dormitory
    main.cpp
)

target_link_libraries(dormitory PRIVATE ftxui::screen ftxui::dom ftxui::component)

target_include_directories(dormitory PRIVATE
    ${CMAKE_SOURCE_DIR}/src
)

# output/dormitory.exe
set_target_properties(dormitory PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_SOURCE_DIR}/output
)

target_compile_definitions(dormitory PRIVATE
    "DATA_DIR_PATH=\"${CMAKE_SOURCE_DIR}/data\""
)
